Quick Start

Start in Seconds

No SDK to install. No keys to manage. Just clean HTTP calls.

terminal
# Free: Get live BTC price
curl https://api.aiservices.to/v1/price/BTC

# Free: Batch prices
curl "https://api.aiservices.to/v1/prices?symbols=BTC,ETH,SOL"

# Free: Fear & Greed Index
curl https://api.aiservices.to/v1/fear-greed

# Paid: Technical indicators (x402 payment)
curl https://api.aiservices.to/v1/indicators/BTC
# Free: Get live BTC price
import requests

response = requests.get("https://api.aiservices.to/v1/price/BTC")
data = response.json()
print(f"BTC: ${data['price_usd']:,.0f}")

# Batch prices
response = requests.get(
    "https://api.aiservices.to/v1/prices",
    params={"symbols": "BTC,ETH,SOL,XRP"}
)
for coin in response.json()["data"]:
    print(f"{coin['symbol']}: ${coin['price_usd']:,.2f}")
// Free: Get live BTC price
const res = await fetch("https://api.aiservices.to/v1/price/BTC");
const data = await res.json();
console.log(`BTC: $${data.price_usd.toLocaleString()}`);

// Batch prices
const batch = await fetch(
  "https://api.aiservices.to/v1/prices?symbols=BTC,ETH,SOL"
).then(r => r.json());

// Fear & Greed
const sentiment = await fetch("https://api.aiservices.to/v1/fear-greed")
  .then(r => r.json());
console.log(`Market sentiment: ${sentiment.label}`);
